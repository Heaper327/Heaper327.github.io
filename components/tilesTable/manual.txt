Manual:
    
    randomChar()
        function:
            generate random character
        arguments:
            none
        return:
            a random character from A-Z . ,

    randomNum(min, max)
        function:
            generate random number within a range
        arguments:
            min: smallest possible return value
            max: largest possible return value
        return:
            a random 

    initTable(id, tileWidthA, tileHeightA, minTimeA, maxTimeA, backColorA, charColorA, switchTimeA, quoteAPI)
        function:
            assign value to global variables, initiate a table
        arguments:
            id: string, id of target canvas
            tileWidthA: number, width of each tile
            tileHeightA: number, height of each tile
            minTimeA: number, the minimum time each tile refreshes
            maxTimeA: number, the maximum time each tile refreshes
            backColorA: string, background color of tiles
            charColorA: string, color of characters in tiles
            switchTimeA: number, length of time between two refreshes
            quoteAPI: function, a function that when called returns a string to display
        return:
            return 0 by default, return -1 when error occurs

    isCoord(x, y){
        function:
            check the legitimacy of a set of coordinate that indicates a tile
        arguments:
            x: number, x coordinate
            y: number, y coordinate
        return:
            null if false, 1 if true

    findTile(x, y)
        function:
            find the absolute coordinate of a tile by its set of coordinate
        arguments:
            x: number, x coordinate
            y: number, y coordinate
        return:
            [x, y] set of absolute coordinate if the set of coordinate is legit, null if not

    setTile(x, y, char)
        function:
            set the inner character of a given tile
        arguments:
            x: number, x coordinate of the tile
            y: number, y coordinate of the tile
            char: char, character to be set inside the given tile
        return:
            0 by default, -1 when failed

    initTile(x, y)
        function:
            initiate a given tile, erase its content and redraw it
        arguments:
            x: number, x coordinate
            y: number, y coordinate
        return:
            0 by default, -1 when failed

    makeMap(str)
        function:
            generate a map that can be used to set the table use the string given. 
        arguments:
            str: string, a string used to generate the map
        return:
            a map
    }

    setTableByMap(map)
        function:
            set table according to a map. If the refresh_times_remain of a tile it is corresponding to in the map is above 0, a random character will be displayed on it. If the refresh_times_remain of a tile in the map is 0, the character it is corresponding to in the map will be displayed on it.
        arguments:
            map: map, a map generated by makeMap 
        return:
            given map that is modified by it

    startTable()
        function:
            generates a map by a string returned by getQuote, and repeatedly call setTableByMap to set table by the map generated
        arguments:
            none
        return:
            0 by default, -1 if failed

    refreshTable()
        function:
            a infinite loop function that set table by setTableByMap, and use the returned map as argument the next time it calls setTableByMap. It stops for a period of time between two calls
        arguments:
            none
        return:
            0

    endTable()
        function:
            end the running refreshTable() function
        arguments:
            none
        return:
            none
